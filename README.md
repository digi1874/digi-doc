```
// 简单例子
import digi, { createData } from 'digi'

// 创建监听数据
const data = createData({ a: 123 })

// 添加元素
digi({ text: data.$tp('a') })

console.log(document.body.lastChild.outerHTML)
// => <div>123</div>

data.a = 321
// => watch a => newVal: 321, oldVal: 123

console.log(document.body.lastChild.outerHTML)
// => <div>321</div>
```
